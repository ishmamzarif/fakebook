Fakebook — Database Schema & SQL Documentation

Project: Fakebook (Social Media Platform)  |  Database: PostgreSQL (Neon)  |  Date: April 2026

1. Database Overview

Fakebook uses PostgreSQL as its relational database. The schema is designed around core social media features: user accounts, friend connections, messaging, content sharing, reactions, and notifications. All timestamps default to CURRENT_TIMESTAMP and sessions are forced to UTC.

2. Users Management

The users table is the central entity. It stores authentication credentials (hashed passwords), profile information, and social metadata. Each user has a unique username, email, and optional profile link.

2.1 Table: users

CREATE TABLE users (
    user_id       SERIAL PRIMARY KEY,
    username      VARCHAR(50) NOT NULL UNIQUE,
    email         VARCHAR(150) NOT NULL UNIQUE,
    password      TEXT NOT NULL,
    full_name     VARCHAR(100) NOT NULL,
    nickname      VARCHAR(50),
    profile_link  TEXT UNIQUE,
    avatar_url    TEXT,
    profile_picture TEXT,
    cover_picture TEXT,
    bio           TEXT,
    phone_number  VARCHAR(20),
    curr_institution VARCHAR(150),
    address       TEXT,
    is_private    BOOLEAN DEFAULT FALSE,
    num_friends   INTEGER DEFAULT 0,
    created_at    TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

2.2 Key Queries

-- Register a new user
INSERT INTO users (username, email, password, full_name)
VALUES ($1, $2, $3, $4) RETURNING *;

-- Login: find user by email
SELECT * FROM users WHERE email = $1;

-- Update profile
UPDATE users SET full_name=$1, bio=$2, profile_picture=$3
WHERE user_id = $4 RETURNING *;

-- Search users
SELECT user_id, username, full_name, profile_picture
FROM users WHERE username ILIKE '%' || $1 || '%'
   OR full_name ILIKE '%' || $1 || '%';

3. Friend System

Friendships go through a request-accept workflow. A user sends a friend_request; the receiver can accept or reject. On acceptance, a row is inserted into the friends table. The blocks table prevents unwanted interactions.

3.1 Table: friend_requests

CREATE TABLE friend_requests (
    request_id  SERIAL PRIMARY KEY,
    sender_id   INT NOT NULL REFERENCES users(user_id) ON DELETE CASCADE,
    receiver_id INT NOT NULL REFERENCES users(user_id) ON DELETE CASCADE,
    status      VARCHAR(20) DEFAULT 'pending',
    created_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE (sender_id, receiver_id)
);

3.2 Table: friends

CREATE TABLE friends (
    friendship_id SERIAL PRIMARY KEY,
    friend1_id    INT NOT NULL REFERENCES users(user_id) ON DELETE CASCADE,
    friend2_id    INT NOT NULL REFERENCES users(user_id) ON DELETE CASCADE,
    created_at    TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CHECK (friend1_id <> friend2_id)
);

3.3 Table: blocks

CREATE TABLE blocks (
    blocker_id INT NOT NULL REFERENCES users(user_id),
    blocked_id INT NOT NULL REFERENCES users(user_id),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (blocker_id, blocked_id)
);

3.4 Key Queries

-- Send friend request
INSERT INTO friend_requests (sender_id, receiver_id) VALUES ($1, $2);

-- Accept friend request
UPDATE friend_requests SET status = 'accepted'
WHERE sender_id = $1 AND receiver_id = $2 AND status = 'pending';

-- Insert into friends table on acceptance
INSERT INTO friends (friend1_id, friend2_id) VALUES ($1, $2);

-- Get all friends of a user
SELECT u.user_id, u.username, u.full_name, u.profile_picture
FROM friends f
JOIN users u ON u.user_id = CASE
    WHEN f.friend1_id = $1 THEN f.friend2_id
    ELSE f.friend1_id END
WHERE f.friend1_id = $1 OR f.friend2_id = $1;

-- Reject / unfriend
DELETE FROM friends WHERE friendship_id = $1;

4. Messaging System

Messaging supports both 1-on-1 and group chats. A conversation acts as a container; members are tracked in conversation_members. Each message belongs to a conversation. Read receipts are stored in message_reads.

4.1 Table: conversations

CREATE TABLE conversations (
    conversation_id SERIAL PRIMARY KEY,
    is_group        BOOLEAN DEFAULT FALSE,
    group_name      TEXT,
    group_photo_url TEXT,
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    created_by      INT NOT NULL REFERENCES users(user_id) ON DELETE CASCADE
);

4.2 Table: conversation_members

CREATE TABLE conversation_members (
    conversation_id INT NOT NULL REFERENCES conversations(conversation_id) ON DELETE CASCADE,
    user_id         INT NOT NULL REFERENCES users(user_id) ON DELETE CASCADE,
    PRIMARY KEY (conversation_id, user_id)
);

4.3 Table: messages

CREATE TABLE messages (
    message_id      SERIAL PRIMARY KEY,
    conversation_id INT NOT NULL REFERENCES conversations(conversation_id) ON DELETE CASCADE,
    sender_id       INT NOT NULL REFERENCES users(user_id) ON DELETE CASCADE,
    content         TEXT NOT NULL,
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

4.4 Table: message_reads

CREATE TABLE message_reads (
    message_id INT NOT NULL REFERENCES messages(message_id) ON DELETE CASCADE,
    user_id    INT NOT NULL REFERENCES users(user_id) ON DELETE CASCADE,
    read_at    TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (message_id, user_id)
);

4.5 Key Queries

-- Create a 1-on-1 conversation
INSERT INTO conversations (is_group, created_by) VALUES (FALSE, $1)
RETURNING conversation_id;
INSERT INTO conversation_members (conversation_id, user_id) VALUES ($1, $2);

-- Send a message
INSERT INTO messages (conversation_id, sender_id, content)
VALUES ($1, $2, $3) RETURNING *;

-- Get conversation list with last message and unread count
SELECT c.*, m.content AS last_message, m.created_at AS last_message_time,
  (SELECT COUNT(*) FROM messages msg
   WHERE msg.conversation_id = c.conversation_id
     AND msg.sender_id != $1
     AND NOT EXISTS (
       SELECT 1 FROM message_reads mr
       WHERE mr.message_id = msg.message_id AND mr.user_id = $1
     )) AS unread_count
FROM conversations c
JOIN conversation_members cm ON cm.conversation_id = c.conversation_id
LEFT JOIN LATERAL (
  SELECT content, created_at FROM messages
  WHERE conversation_id = c.conversation_id
  ORDER BY created_at DESC LIMIT 1
) m ON TRUE
WHERE cm.user_id = $1
ORDER BY m.created_at DESC NULLS LAST;

-- Mark messages as read
INSERT INTO message_reads (message_id, user_id)
SELECT message_id, $2 FROM messages
WHERE conversation_id = $1 AND sender_id != $2
ON CONFLICT DO NOTHING;

5. Posts & Content

Posts can be text-only (post_type = 'c') or include media ('p'). Media files are stored in Cloudinary and referenced via the content_media table. Users can tag friends when creating posts using the post_tags table.

5.1 Table: posts

CREATE TABLE posts (
    post_id    SERIAL PRIMARY KEY,
    user_id    INT NOT NULL REFERENCES users(user_id) ON DELETE CASCADE,
    caption    TEXT,
    visibility INT DEFAULT 1,
    post_type  VARCHAR(10) DEFAULT 'c',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

5.2 Table: content_media

CREATE TABLE content_media (
    media_id    SERIAL PRIMARY KEY,
    media_url   TEXT NOT NULL,
    media_type  VARCHAR(10),       -- 'image' or 'video'
    type        VARCHAR(50) NOT NULL, -- 'post', 'message', 'story'
    reference_id INTEGER,          -- post_id, message_id, etc.
    media_order INT,
    created_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

Design Note: content_media is a polymorphic table. The type column indicates what entity the media belongs to, and reference_id is the FK to that entity. This avoids creating separate media tables for posts, messages, and stories.

5.3 Table: post_tags

CREATE TABLE post_tags (
    post_id        INT NOT NULL REFERENCES posts(post_id) ON DELETE CASCADE,
    tagged_user_id INT NOT NULL REFERENCES users(user_id) ON DELETE CASCADE,
    PRIMARY KEY (post_id, tagged_user_id)
);

5.4 Key Queries

-- Create a post
INSERT INTO posts (user_id, caption, post_type)
VALUES ($1, $2, $3) RETURNING *;

-- Tag users in a post
INSERT INTO post_tags (post_id, tagged_user_id) VALUES ($1, $2);

-- Get feed (own posts + friends' posts) with media, reactions, tags
SELECT p.post_id, p.user_id, p.caption, p.post_type, p.created_at,
  u.username, u.full_name, u.profile_picture,
  COUNT(DISTINCT CASE WHEN l.target_type LIKE 'post_reaction:%'
    THEN l.like_id END)::int AS likes_count,
  COUNT(DISTINCT c.comment_id)::int AS comments_count,
  COALESCE(JSON_AGG(JSON_BUILD_OBJECT(
    'media_id', cm.media_id, 'media_url', cm.media_url,
    'media_type', cm.media_type, 'media_order', cm.media_order
  )) FILTER (WHERE cm.media_id IS NOT NULL), '[]'::json) AS media,
  (SELECT COALESCE(JSON_AGG(JSON_BUILD_OBJECT(
    'user_id', ut.user_id, 'username', ut.username, 'full_name', ut.full_name
  )), '[]'::json) FROM post_tags pt
   JOIN users ut ON pt.tagged_user_id = ut.user_id
   WHERE pt.post_id = p.post_id) AS tags
FROM posts p
JOIN users u ON p.user_id = u.user_id
LEFT JOIN likes l ON l.target_id = p.post_id
LEFT JOIN comments c ON c.post_id = p.post_id
LEFT JOIN content_media cm ON cm.type='post' AND cm.reference_id=p.post_id
WHERE p.user_id = $1 OR p.user_id IN (
  SELECT friend1_id FROM friends WHERE friend2_id = $1
  UNION
  SELECT friend2_id FROM friends WHERE friend1_id = $1)
GROUP BY p.post_id, u.user_id
ORDER BY p.created_at DESC;

6. Comments & Reactions

6.1 Table: comments

CREATE TABLE comments (
    comment_id SERIAL PRIMARY KEY,
    post_id    INT NOT NULL REFERENCES posts(post_id) ON DELETE CASCADE,
    user_id    INT NOT NULL REFERENCES users(user_id) ON DELETE CASCADE,
    content    TEXT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

6.2 Table: comment_replies

CREATE TABLE comment_replies (
    reply_id   SERIAL PRIMARY KEY,
    comment_id INT NOT NULL REFERENCES comments(comment_id) ON DELETE CASCADE,
    user_id    INT NOT NULL REFERENCES users(user_id) ON DELETE CASCADE,
    content    TEXT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

6.3 Table: likes (Polymorphic Reactions)

CREATE TABLE likes (
    like_id     SERIAL PRIMARY KEY,
    user_id     INT NOT NULL REFERENCES users(user_id) ON DELETE CASCADE,
    target_type TEXT NOT NULL,  -- e.g. 'post_reaction:❤️', 'comment_like'
    target_id   INT NOT NULL,
    created_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

Design Note: The target_type stores both the entity type and the emoji reaction (e.g., post_reaction:👍). This allows multiple reaction types on a single post without needing separate columns.

8. Notifications System TRIGGER-BASED

Notifications are generated automatically via PostgreSQL triggers. When a user sends a message, comments, likes, creates a post, or tags someone, a trigger function inserts the appropriate notification row. The frontend fetches these and displays them with actor details.

Key columns: is_read (grey highlight until "Mark as read"), is_seen (badge count, cleared on visiting the page), actor_id (who triggered the notification).

8.1 Table: notifications (with added columns)

CREATE TABLE notifications (
    notification_id SERIAL PRIMARY KEY,
    user_id         INT NOT NULL REFERENCES users(user_id) ON DELETE CASCADE,
    actor_id        INT REFERENCES users(user_id) ON DELETE CASCADE,
    type            VARCHAR(50) NOT NULL,
    reference_id    INT,
    is_read         BOOLEAN DEFAULT FALSE,
    is_seen         BOOLEAN DEFAULT FALSE,
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

9. Database Triggers — Complete Code

9.1 Friend Request Sent

CREATE OR REPLACE FUNCTION notify_friend_request_sent()
RETURNS TRIGGER AS $$
BEGIN
  INSERT INTO notifications (user_id, actor_id, type, reference_id)
  VALUES (NEW.receiver_id, NEW.sender_id, 'friend_request', NEW.request_id);
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_friend_request_sent
AFTER INSERT ON friend_requests
FOR EACH ROW
EXECUTE FUNCTION notify_friend_request_sent();

9.2 Friend Request Accepted

CREATE OR REPLACE FUNCTION notify_friend_request_accepted()
RETURNS TRIGGER AS $$
BEGIN
  IF OLD.status = 'pending' AND NEW.status = 'accepted' THEN
    INSERT INTO notifications (user_id, actor_id, type, reference_id)
    VALUES (NEW.sender_id, NEW.receiver_id, 'friend_request_accepted', NEW.request_id);
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_friend_request_accepted
AFTER UPDATE ON friend_requests
FOR EACH ROW
EXECUTE FUNCTION notify_friend_request_accepted();

9.3 Message Sent

CREATE OR REPLACE FUNCTION notify_message_sent()
RETURNS TRIGGER AS $$
BEGIN
  INSERT INTO notifications (user_id, actor_id, type, reference_id)
  SELECT user_id, NEW.sender_id, 'message', NEW.message_id
  FROM conversation_members
  WHERE conversation_id = NEW.conversation_id
    AND user_id != NEW.sender_id;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_message_sent
AFTER INSERT ON messages
FOR EACH ROW
EXECUTE FUNCTION notify_message_sent();

9.4 Comment Added (also notifies tagged users)

CREATE OR REPLACE FUNCTION notify_comment_added()
RETURNS TRIGGER AS $$
DECLARE
  post_owner_id INTEGER;
BEGIN
  SELECT user_id INTO post_owner_id FROM posts WHERE post_id = NEW.post_id;

  -- Notify post owner (if commenter is not the owner)
  IF post_owner_id IS NOT NULL AND post_owner_id != NEW.user_id THEN
    INSERT INTO notifications (user_id, actor_id, type, reference_id)
    VALUES (post_owner_id, NEW.user_id, 'comment', NEW.comment_id);
  END IF;

  -- Also notify users tagged in the post
  INSERT INTO notifications (user_id, actor_id, type, reference_id)
  SELECT pt.tagged_user_id, NEW.user_id, 'comment', NEW.comment_id
  FROM post_tags pt
  WHERE pt.post_id = NEW.post_id
    AND pt.tagged_user_id != NEW.user_id
    AND pt.tagged_user_id != post_owner_id;

  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_comment_added
AFTER INSERT ON comments
FOR EACH ROW
EXECUTE FUNCTION notify_comment_added();

9.5 Comment Reply Added

CREATE OR REPLACE FUNCTION notify_comment_reply_added()
RETURNS TRIGGER AS $$
DECLARE
  parent_commenter_id INTEGER;
BEGIN
  SELECT user_id INTO parent_commenter_id
  FROM comments WHERE comment_id = NEW.comment_id;

  IF parent_commenter_id IS NOT NULL AND parent_commenter_id != NEW.user_id THEN
    INSERT INTO notifications (user_id, actor_id, type, reference_id)
    VALUES (parent_commenter_id, NEW.user_id, 'comment_reply', NEW.reply_id);
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_comment_reply_added
AFTER INSERT ON comment_replies
FOR EACH ROW
EXECUTE FUNCTION notify_comment_reply_added();

9.6 Like / React Added (also notifies tagged users)

CREATE OR REPLACE FUNCTION notify_like_added()
RETURNS TRIGGER AS $$
DECLARE
  target_owner_id INTEGER;
  target_post_id  INTEGER;
BEGIN
  IF NEW.target_type LIKE 'post%' THEN
    SELECT user_id, post_id INTO target_owner_id, target_post_id
    FROM posts WHERE post_id = NEW.target_id;
  ELSIF NEW.target_type LIKE 'comment%' THEN
    SELECT c.user_id, c.post_id INTO target_owner_id, target_post_id
    FROM comments c WHERE c.comment_id = NEW.target_id;
  END IF;

  -- Notify content owner
  IF target_owner_id IS NOT NULL AND target_owner_id != NEW.user_id THEN
    INSERT INTO notifications (user_id, actor_id, type, reference_id)
    VALUES (target_owner_id, NEW.user_id, 'like', NEW.like_id);
  END IF;

  -- Notify tagged users on the post (for post likes only)
  IF NEW.target_type LIKE 'post%' AND target_post_id IS NOT NULL THEN
    INSERT INTO notifications (user_id, actor_id, type, reference_id)
    SELECT pt.tagged_user_id, NEW.user_id, 'like', NEW.like_id
    FROM post_tags pt
    WHERE pt.post_id = target_post_id
      AND pt.tagged_user_id != NEW.user_id
      AND pt.tagged_user_id != target_owner_id;
  END IF;

  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_like_added
AFTER INSERT ON likes
FOR EACH ROW
EXECUTE FUNCTION notify_like_added();

9.7 New Post Created (notify all friends)

CREATE OR REPLACE FUNCTION notify_post_to_friends()
RETURNS TRIGGER AS $$
BEGIN
  INSERT INTO notifications (user_id, actor_id, type, reference_id)
  SELECT
    CASE WHEN friend1_id = NEW.user_id THEN friend2_id
         ELSE friend1_id END,
    NEW.user_id, 'new_post', NEW.post_id
  FROM friends
  WHERE friend1_id = NEW.user_id OR friend2_id = NEW.user_id;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_post_created
AFTER INSERT ON posts
FOR EACH ROW
EXECUTE FUNCTION notify_post_to_friends();

9.8 User Tagged in Post

CREATE OR REPLACE FUNCTION notify_tagged_user()
RETURNS TRIGGER AS $$
DECLARE
  author_id INTEGER;
BEGIN
  SELECT user_id INTO author_id FROM posts WHERE post_id = NEW.post_id;
  IF author_id IS NOT NULL THEN
    INSERT INTO notifications (user_id, actor_id, type, reference_id)
    VALUES (NEW.tagged_user_id, author_id, 'post_tag', NEW.post_id);
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_tag_added
AFTER INSERT ON post_tags
FOR EACH ROW
EXECUTE FUNCTION notify_tagged_user();

10. Notification Fetch Query

The API resolves navigation targets (post_id, conversation_id) so each notification is clickable on the frontend.

SELECT n.notification_id, n.user_id, n.type, n.reference_id,
  n.created_at, n.is_read, n.is_seen, n.actor_id,
  u.username AS actor_username,
  u.full_name AS actor_full_name,
  u.profile_picture AS actor_profile_picture,
  -- Resolve target post_id
  CASE
    WHEN n.type IN ('new_post','post_tag') THEN n.reference_id
    WHEN n.type = 'comment' THEN
      (SELECT c.post_id FROM comments c WHERE c.comment_id = n.reference_id LIMIT 1)
    WHEN n.type = 'comment_reply' THEN
      (SELECT c.post_id FROM comment_replies r JOIN comments c ON c.comment_id = r.comment_id
       WHERE r.reply_id = n.reference_id LIMIT 1)
    WHEN n.type = 'like' THEN
      (SELECT CASE WHEN l.target_type LIKE 'post%' THEN l.target_id
        WHEN l.target_type LIKE 'comment%' THEN
          (SELECT c.post_id FROM comments c WHERE c.comment_id = l.target_id LIMIT 1)
       END FROM likes l WHERE l.like_id = n.reference_id LIMIT 1)
  END AS post_id,
  -- Resolve conversation_id for messages  
  CASE WHEN n.type = 'message' THEN
    (SELECT m.conversation_id FROM messages m WHERE m.message_id = n.reference_id LIMIT 1)
  END AS conversation_id
FROM notifications n
LEFT JOIN users u ON n.actor_id = u.user_id
WHERE n.user_id = $1
ORDER BY n.created_at DESC;

WHERE user_id = $1 AND is_seen = FALSE;

-- Mark all notifications as "read" (removes grey highlight)

WHERE user_id = $1 AND is_read = FALSE;
